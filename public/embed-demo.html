Use our JavaScript SDK for more control and customization:

‚öõÔ∏è Method 4: React Component

For React applications, use this component approach:

import React, { useEffect } from 'react'; const FormPulseEmbed = ({ formId, theme = 'light', primaryColor = '#3b82f6', onSubmit, onLoad, onError }) => { useEffect(() => { // Load the embed script const script = document.createElement('script'); script.src = 'http://localhost:5173/embed-script.js'; script.onload = () => { window.FormPulseEmbed.init({ formId, container: `#formpulse-${formId}`, theme, primaryColor, onSubmit, onLoad, onError }); }; document.head.appendChild(script); return () => { document.head.removeChild(script); }; }, [formId, theme, primaryColor]); return <div id={`formpulse-${formId}`} />; }; // Usage <FormPulseEmbed formId="demo-1" theme="light" primaryColor="#8b5cf6" onSubmit={(data) => console.log('Submitted:', data)} onLoad={() => console.log('Form loaded')} onError={(error) => console.error('Error:', error)} />
